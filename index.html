// 👇 1. 您的管理者 Email (接收新預約/取消通知用) 👇 var ADMIN_EMAIL = 'your-email@gmail.com'; // 👇 2. 新預約要「寫入」哪一個特定日曆？ 👇 var TARGET_CALENDAR_ID = 'your-email@gmail.com'; function setup() { CalendarApp.getAllCalendars(); GmailApp.getInboxUnreadCount(); } function doGet(e) { try { return processAction(e.parameter); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ error: error.toString() })).setMimeType(ContentService.MimeType.JSON).setHeader('Access-Control-Allow-Origin', '*'); } } function doPost(e) { try { var data = {}; if (e.postData && e.postData.contents) { data = JSON.parse(e.postData.contents); } else { data = e.parameter; } return processAction(data); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ error: error.toString() })).setMimeType(ContentService.MimeType.JSON).setHeader('Access-Control-Allow-Origin', '*'); } } function processAction(data) { var action = data.action; // 【情境一】：取得行事曆忙碌時間 if (action === 'getBusy') { var now = new Date(); var end = new Date(); end.setDate(now.getDate() + 30); var allBusySlots = []; var calendars = CalendarApp.getAllCalendars(); for (var i = 0; i < calendars.length; i++) { var calId = calendars[i].getId(); // 💡 過濾節慶與生日 if (calId.indexOf('#holiday@') !== -1 || calId.indexOf('#contacts@') !== -1) { continue; } // 💡 加強防護：把單一日曆的讀取包起來，失敗了也只影響這個日曆 try { var events = calendars[i].getEvents(now, end); for (var j = 0; j < events.length; j++) { var eventTitle = events[j].getTitle(); if (events[j].isAllDayEvent() && eventTitle.indexOf('生日') !== -1) { continue; } allBusySlots.push({ start: events[j].getStartTime().toISOString(), end: events[j].getEndTime().toISOString() }); } } catch(e) { // 如果這個日曆讀取失敗，我們就靜靜地跳過它，繼續讀下一個 // console.error('無法讀取日曆 ' + calId + ': ' + e); } } return ContentService.createTextOutput(JSON.stringify({ busy: allBusySlots })).setMimeType(ContentService.MimeType.JSON).setHeader('Access-Control-Allow-Origin', '*'); } var name = data.name || '未知'; var email = data.email || ''; var phone = data.phone || ''; var lineName = data.lineName || ''; var dateRaw = data.dateRaw || ''; var dateLabel = data.dateLabel || ''; var time = data.time || ''; var notes = data.notes || '無'; var targetCalendar = null; try { targetCalendar = CalendarApp.getCalendarById(TARGET_CALENDAR_ID); } catch(err) {} if (!targetCalendar) targetCalendar = CalendarApp.getDefaultCalendar(); // 🛑 處理「取消預約」 if (action === 'cancel') { var startTimeStr = dateRaw + 'T' + time + ':00+08:00'; var startTime = new Date(startTimeStr); var endTime = new Date(startTime.getTime() + 3600000); var calendarEvents = targetCalendar.getEvents(startTime, endTime); for (var k = 0; k < calendarEvents.length; k++) { var desc = calendarEvents[k].getDescription() || ''; var title = calendarEvents[k].getTitle() || ''; if (desc.indexOf(email) !== -1 || desc.indexOf(phone) !== -1 || title.indexOf(name) !== -1) { calendarEvents[k].deleteEvent(); break; } } var adminCancelBody = "預約者：" + name + "\n電話：" + phone + "\n信箱：" + email + "\n\n已取消 " + dateRaw + " " + time + " 的協談。"; GmailApp.sendEmail(ADMIN_EMAIL, "【取消通知】" + name + " 已取消協談", adminCancelBody); if (email) { var userCancelBody = "親愛的 " + name + "：\n\n您已成功取消 " + dateRaw + " " + time + " 的線上協談。\n歡迎您未來有需要時再次預約！"; GmailApp.sendEmail(email, "【取消成功】線上協談取消通知", userCancelBody); } return ContentService.createTextOutput("Cancel Success").setHeader('Access-Control-Allow-Origin', '*'); } // ✅ 處理「新增預約」 if (action === 'book') { var startTimeStr2 = dateRaw + 'T' + time + ':00+08:00'; var startTime2 = new Date(startTimeStr2); var endTime2 = new Date(startTime2.getTime() + 3600000); targetCalendar.createEvent("協談預約 - " + name, startTime2, endTime2, { description: "預約者：" + name + "\nLine：" + lineName + "\n電話：" + phone + "\n信箱：" + email + "\n備註：\n" + notes }); var adminBookBody = "日期：" + dateLabel + "\n時間：" + time + "\n姓名：" + name + "\nLine：" + lineName + "\n電話：" + phone + "\n信箱：" + email + "\n備註：" + notes; GmailApp.sendEmail(ADMIN_EMAIL, "【新預約通知】" + name + " 新增了協談線上預約", adminBookBody); if (email) { var userBookBody = "親愛的 " + name + " 您好：\n\n您已成功預約線上協談！以下是您的預約資訊：\n\n日期：" + dateLabel + "\n時間：" + time + "\n\n若需取消，請至預約網站輸入您的 Email 與手機號碼即可取消。"; GmailApp.sendEmail(email, "【預約成功通知】您的協談線上預約已確認", userBookBody); } return ContentService.createTextOutput("Book Success").setHeader('Access-Control-Allow-Origin', '*'); } return ContentService.createTextOutput("OK").setHeader('Access-Control-Allow-Origin', '*'); }
